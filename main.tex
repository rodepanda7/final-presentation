  \section{Introduction}
  \begin{frame}{Call-by-name}
    \begin{block}{Let f be defined as}
      \begin{flalign*}
        & f (x) = x * x &
      \end{flalign*}
    \end{block} \pause

    \begin{block}{Call-by-name evaluation of f (3 + 3)}
      \begin{align*}
        \onslide<2->{f (3 + 3) & \to (3 + 3) * (3 + 3)} \\
        \onslide<3->{          & \to 6 * (3 + 3)} \\
        \onslide<4->{          & \to 6 * 6} \\
        \onslide<5->{          & \to 36}
      \end{align*}
    \end{block} \pause
  \end{frame}

  \begin{frame}{Call-by-value}
    \begin{block}{Let f be defined as}
      \begin{flalign*}
        & f (x) = x * x &
      \end{flalign*}
    \end{block} \pause

    \begin{block}{Call-by-value evaluation of f (3 + 3)}
      \begin{align*}
        \onslide<2->{f (3 + 3) & \to f (6) } \\
        \onslide<3->{          & \to 6 * 6} \\
        \onslide<4->{          & \to 36}
      \end{align*}
    \end{block} \pause
  \end{frame}
